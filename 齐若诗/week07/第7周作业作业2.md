# 3. 核心提取函数
# ==========================================
def extract_information(user_input: str) -> dict:
    try:
        response = client.chat.completions.create(
            model="qwen-plus", # 可根据需求切换为 qwen-max
            messages=[
                {"role": "system", "content": SYSTEM_PROMPT},
                {"role": "user", "content": f'User: "{user_input}"\nAssistant:'}
            ],
            temperature=0.1, # 降低温度，保证信息抽取任务的确定性和稳定性
            top_p=0.8,
        )
        
        # 获取返回内容
        result_text = response.choices[0].message.content.strip()
        
        # 清洗可能存在的 Markdown 标记 (大模型有时会调皮地加上 ```json )
        result_text = re.sub(r'^```json\s*', '', result_text)
        result_text = re.sub(r'\s*```$', '', result_text)
        
        # 将字符串解析为 Python 字典
        result_dict = json.loads(result_text)
        return result_dict
        
    except json.JSONDecodeError as e:
        print(f"JSON 解析失败: {e}\n大模型原始返回: {result_text}")
        return {"domain": "Unknown", "intent": "Unknown", "slots": {}}
    except Exception as e:
        print(f"API 调用发生错误: {e}")
        return {"domain": "Error", "intent": "Error", "slots": {}}
